users table (primary = email)
  email char(1000)
  first_name char(1000)
  last_name char(1000)
  password char(1000)
  salt char(1000)

ratings table (primary = user, route_key)
  user char(1000)
  route_key char(1000)
  safety (INT)
  efficiency (INT)
  scenery (INT)

routes table (primary = user, route_key)
  user char(1000)
  route_key char(1000)
  name char(1000)
  from char(1000)
  to char(1000)
  order (INT)

tips_and_accidents table (primary = user, type, id)
  user char(1000)
  type = INT
  id char(1000)
  comment char(1000)
  x char(1000)
  y char(1000)
